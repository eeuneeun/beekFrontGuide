function customSelect() {
    var x, i, j, l, ll, selElmnt, a, b, c;

    x = document.getElementsByClassName("custom_select");
    l = x.length;
    for (i = 0; i < l; i++) {
    selElmnt = x[i].getElementsByTagName("select")[0];
    ll = selElmnt.length;

    a = document.createElement("div");
    a.setAttribute("class", "select_selected");
    a.innerHTML = selElmnt.options[selElmnt.selectedIndex].innerHTML;
    x[i].appendChild(a);

    b = document.createElement("div");
    b.setAttribute("class", "select_items select_hide custom_scrollbar");

    for (j = 0; j < ll; j++) {
        c = document.createElement("div");
        c.innerHTML = selElmnt.options[j].innerHTML;

        c.addEventListener("click", function (e) {
        var y, i, k, s, h, sl, yl;

        s = this.parentNode.parentNode.getElementsByTagName("select")[0];
        sl = s.length;
        h = this.parentNode.previousSibling;

        for (i = 0; i < sl; i++) {
            if (s.options[i].innerHTML == this.innerHTML) {
            s.selectedIndex = i;
            h.innerHTML = this.innerHTML;
            y = this.parentNode.getElementsByClassName("same-as-selected");
            yl = y.length;

            for (k = 0; k < yl; k++) {
                y[k].removeAttribute("class");
            }

            this.setAttribute("class", "same-as-selected");
            break;
            }
        }

        h.click();
        });

        b.appendChild(c);
    }

    x[i].appendChild(b);

    a.addEventListener("click", function (e) {
        e.stopPropagation();
        closeAllSelect(this);
        this.nextSibling.classList.toggle("select_hide");
        this.classList.toggle("select_arrow_active");
    });
    }

    function closeAllSelect(elmnt) {
    /*a function that will close all select boxes in the document,
    except the current select box:*/
    var x,
        y,
        i,
        xl,
        yl,
        arrNo = [];
    x = document.getElementsByClassName("select_items");
    y = document.getElementsByClassName("select_selected");
    xl = x.length;
    yl = y.length;
    for (i = 0; i < yl; i++) {
        if (elmnt == y[i]) {
        arrNo.push(i);
        } else {
        y[i].classList.remove("select_arrow_active");
        }
    }
    for (i = 0; i < xl; i++) {
        if (arrNo.indexOf(i)) {
        x[i].classList.add("select_hide");
        }
    }
    }

    document.addEventListener("click", closeAllSelect);
}

    5. JS
      /*
* ★ COMMON - Modal
*
*    #target : string
*     - 모달로 사용하고자 하는 요소
*     - querySelector 의 인자로 사용됨
*     - 검색이 될 수 있도록 '' 로 감싸서 제공하면 됨
*     - ex) '#topModalWrap', '.div .modal_wrap', 'input+div' 등등
*
*    #action : string - ['open', 'close', 'toggle']
*     - 타겟이 되는 모달 요소에 주고자 하는 변화
*     - 제공되는 세 가지 문자열 중 하나를 선택하여 유연하게 사용
*
*/
function commonModalHandler(target, action) {
    const targetPanel = document.querySelector(target);
    console.log(targetPanel);

    targetPanel && action == "open"
    ? targetPanel.classList.add("active")
    : action == "close"
    ? targetPanel.classList.remove("active")
    : action == "toggle"
    ? targetPanel.classList.toggle("active")
    : action == "delete" && targetPanel.remove();
}

function tabMenuHandler(e) {

    const tabMenus = tabWrap.querySelectorAll(".tab_menu > li");

        tabMenus[idx].classList.add("active");
    }
    });
}

// gnb 효과
function gnbHandler(e) {
    const wrapper = document.getElementById("wrapper");

    const scrollY = e.target.scrollTop;
    const widthX = window.innerWidth;

    if (scrollY > 70) {
    widthX > 768
        ? wrapper.classList.add("black")
        : wrapper.classList.remove("black");
    wrapper.classList.add("hide_gnb");
    } else if (scrollY <= 70) {
    wrapper.classList.remove("black");
    wrapper.classList.remove("hide_gnb");
    }

    if (widthX >= 768) {
    wrapper.classList.remove("black");
    }
}
